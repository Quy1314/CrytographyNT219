1. RSA
// Base64 form
C:\Tool\openssl-3.5.4\openssl-openssl-3.5.4\gcc\bin\openssl.exe genrsa -out rsaprivate.pem 3072
C:\Tool\openssl-3.5.4\openssl-openssl-3.5.4\gcc\bin\openssl.exe rsa -in rsaprivate.pem -pubout -out rsapubkey.pem

// Binary form
C:\Tool\openssl-3.5.4\openssl-openssl-3.5.4\gcc\bin\openssl.exe rsa -in rsaprivate.pem -outform DER -out rsa-private.der
C:\Tool\openssl-3.5.4\openssl-openssl-3.5.4\gcc\bin\openssl.exe rsa -in rsaprivate.pem -outform DER -pubout -out rsa-pubkey.der

Read:
C:\Tool\openssl-3.5.4\openssl-openssl-3.5.4\gcc\bin\openssl.exe rsa -inform pem -in rsaprivate.pem -text -noout
C:\Tool\openssl-3.5.4\openssl-openssl-3.5.4\gcc\bin\openssl.exe rsa -inform pem -pubin -in rsapubkey.pem -text -noout


// Cerificate
C:\Tool\openssl-3.5.4\openssl-openssl-3.5.4\gcc\bin\openssl.exe req -key rsaprivate.pem -new -out temp.csr
C:\Tool\openssl-3.5.4\openssl-openssl-3.5.4\gcc\bin\openssl.exe req -text -noout -verify -in temp.csr
(may selft-sign the cert in case yous are the trusted rootca)
C:\Tool\openssl-3.5.4\openssl-openssl-3.5.4\gcc\bin\openssl.exe x509 -signkey rsaprivate.pem -in temp.csr -req -days 365 -out uit.crt
C:\Tool\openssl-3.5.4\openssl-openssl-3.5.4\gcc\bin\openssl.exe x509 -in uit.crt -inform pem -text -noout
may use: certutil.exe

2. ECC
C:\Tool\openssl-3.5.4\openssl-openssl-3.5.4\gcc\bin\openssl.exe ecparam -list_curves

C:\Tool\openssl-3.5.4\openssl-openssl-3.5.4\gcc\bin\openssl.exe ecparam -name prime256v1 -genkey -noout -out ec-private-key.pem

C:\Tool\openssl-3.5.4\openssl-openssl-3.5.4\gcc\bin\openssl.exe ec -in ec-private-key.pem -pubout -out ec-public-key.pem

Create
C:\Tool\openssl-3.5.4\openssl-openssl-3.5.4\gcc\bin\openssl.exe req -key ec-private-key.pem -new -out ec-temp.csr
Check
C:\Tool\openssl-3.5.4\openssl-openssl-3.5.4\gcc\bin\openssl.exe req -text -noout -verify -in ec-temp.csr
